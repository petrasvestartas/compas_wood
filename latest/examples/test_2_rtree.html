2_rtree

  • Description: Find neighbors of polyline pairs using the R-Tree search

  • Input: a list of polylines (the list of top and bottom outlines are stored sequentially (0-top, 0-bottom, 1-top, 1-bottom â€¦ N-top, N-bottom))

  • Output: Tuple of 1) a nested list of neighbours 2) AABB box meshes, 3) OBB box meshes

  • Tests: for testing purpose the polylines are stored in compas_wood.data files. Input and output geomety is visualized below.

Data-set: ss_0

# compas_wood
from compas_wood.joinery import rtree
import data_set_plates

# viewer
from compas_wood.viewer_helpers import display

# ==============================================================================
# Get Adjancency between Elements using RTree Search
# ==============================================================================


def test_rtree(boxes_AABB_or_boxes_OOBB=False):

    # Get a list of polyline pairs
    input = data_set_plates.ss_24()

    # Compute Rtree
    neighbours, boxes_AABB, boxes_OOBB = rtree(input)

    # get aabb or oobb boxes
    selected_id = 22
    boxes_selected = []
    boxes = boxes_AABB if boxes_AABB_or_boxes_OOBB else boxes_OOBB

    # get first and its neighbors
    boxes_selected.append(boxes[selected_id])
    for i in neighbours[selected_id][0]:
        boxes_selected.append(boxes[i])

    # Display via Compas_View2
    display(input, None, boxes_selected, 0.01, 0, 0, 0, True)

    # output
    return neighbours


# ==============================================================================
# call the compas_wood methods
# ==============================================================================
test_rtree(True)
